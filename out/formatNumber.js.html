const { TRANSLATION } = require('./constants');


/**
 * Formats a number and print it out
 * @param {string} number - number to be formatter
 * 
 * @example
 * node formatNumbers.js 123
 * @example 
 * node formatNumbers.js 12.3
 * @example 
 * node formatNumbers.js 189406549406232154897922065454630321654651
 */
outputFormattedNumber = number => {
  // we translate it
  let result = '';
  let numberChars = process.stdout.columns;
  // the calcul includes + 1 to eliminate the space character, ' ', if there is not enough space at the end of the line 
  let charsPerLine = Math.floor((numberChars + 1) / 4);
  let numberLines = Math.floor(number.length / charsPerLine) + 1;

  // browsing lines
  for (let i = 0; i < numberLines; i++) {
    // browsing the 3 lines of a character
    for (let j = 0; j < 3; j++) {
      // displaying characters in the current line
      for (let k = 0; k < charsPerLine && i * charsPerLine + k < number.length; k++) {
        result += TRANSLATION[number[i * charsPerLine + k]][j] + ' ';
      }
      result = result.substring(0, result.length - 1) + '\n';
    }
    result += '\n';
  }
  result = result.substring(0, result.length - 1);
  console.log(result);
}

if(process.argv.length < 3) {
  // less than 3 arguments
  console.error('Too few arguments, launch `node formatNumber.js -h` to see the exact syntax');
} else if(process.argv.length > 3) {
  // more than 3 arguments
  console.error('Too many arguments, launch `node formatNumber.js -h` to see the exact syntax');
} else if(process.argv[2] === '-h') {
  // help 
  console.log('\nFormats a number into the corresponding characters, if the screen width is not enough the number will be outputted on multiples lines');
  console.log('Usage: ');
  console.log('\tnode formatNumber.js numberToFormat');
  console.log('numberToFormat is the number that will be formatted on the output, can be integer or float\n');
} else if(isNaN(process.argv[2])) {
  // not a valid entered
  console.error('Problem parameter is not a number');
} else {
  outputFormattedNumber(process.argv[2])
}
